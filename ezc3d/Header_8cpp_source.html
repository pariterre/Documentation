93 void ezc3d::Header::write(std::fstream &f) const
94 {
95  // write the checksum byte and the start point of header
96  int parameterAddessDefault(2);
97  f.write(reinterpret_cast<const char*>(&parameterAddessDefault), ezc3d::BYTE);
98  int checksum(0x50);
99  f.write(reinterpret_cast<const char*>(&checksum), ezc3d::BYTE);
100 
101  // Number of data
102  f.write(reinterpret_cast<const char*>(&_nb3dPoints), 1*ezc3d::DATA_TYPE::WORD);
103  f.write(reinterpret_cast<const char*>(&_nbAnalogsMeasurement), 1*ezc3d::DATA_TYPE::WORD);
104 
105  // Idx of first and last frame
106  size_t firstFrame(_firstFrame + 1); // 1-based!
107  size_t lastFrame(_lastFrame + 1); // 1-based!
108  if (lastFrame > 0xFFFF)
109  lastFrame = 0xFFFF; // Combine this with group("POINT").parameter("FRAMES") = -1
110  f.write(reinterpret_cast<const char*>(&firstFrame), 1*ezc3d::DATA_TYPE::WORD);
111  f.write(reinterpret_cast<const char*>(&lastFrame), 1*ezc3d::DATA_TYPE::WORD);
112 
113  // Some info
114  f.write(reinterpret_cast<const char*>(&_nbMaxInterpGap), 1*ezc3d::DATA_TYPE::WORD);
115  f.write(reinterpret_cast<const char*>(&_scaleFactor), 2*ezc3d::DATA_TYPE::WORD);
116 
117  // Parameters of analog data
118  f.write(reinterpret_cast<const char*>(&_dataStart), 1*ezc3d::DATA_TYPE::WORD);
119  f.write(reinterpret_cast<const char*>(&_nbAnalogByFrame), 1*ezc3d::DATA_TYPE::WORD);
120  float frameRate(_frameRate);
121  f.write(reinterpret_cast<const char*>(&frameRate), 2*ezc3d::DATA_TYPE::WORD);
122  for (int i=0; i<135; ++i)
123  f.write(reinterpret_cast<const char*>(&_emptyBlock1), 1*ezc3d::DATA_TYPE::WORD);
124 
125  // Parameters of keys
126  f.write(reinterpret_cast<const char*>(&_keyLabelPresent), 1*ezc3d::DATA_TYPE::WORD);
127  f.write(reinterpret_cast<const char*>(&_firstBlockKeyLabel), 1*ezc3d::DATA_TYPE::WORD);
128  f.write(reinterpret_cast<const char*>(&_fourCharPresent), 1*ezc3d::DATA_TYPE::WORD);
129 
130  // Parameters of events
131  f.write(reinterpret_cast<const char*>(&_nbEvents), 1*ezc3d::DATA_TYPE::WORD);
132  f.write(reinterpret_cast<const char*>(&_emptyBlock2), 1*ezc3d::DATA_TYPE::WORD);
133  for (unsigned int i = 0; i < _eventsTime.size(); ++i)
134  f.write(reinterpret_cast<const char*>(&_eventsTime[i]), 2*ezc3d::DATA_TYPE::WORD);
135  for (unsigned int i = 0; i < _eventsDisplay.size(); ++i)
136  f.write(reinterpret_cast<const char*>(&_eventsDisplay[i]), 1*ezc3d::DATA_TYPE::WORD);
137  f.write(reinterpret_cast<const char*>(&_emptyBlock3), 1*ezc3d::DATA_TYPE::WORD);
138  for (unsigned int i = 0; i < _eventsLabel.size(); ++i){
139  const char* event = _eventsLabel[i].c_str();
140  f.write(event, 2*ezc3d::DATA_TYPE::WORD);
141  }
142  for (int i=0; i<22; ++i)
143  f.write(reinterpret_cast<const char*>(&_emptyBlock4), 1*ezc3d::DATA_TYPE::WORD);
144 }

146 void ezc3d::Header::read(ezc3d::c3d &c3d, std::fstream &file)
147 {
148  // Parameter address
149  _parametersAddress = c3d.readUint(file, 1*ezc3d::DATA_TYPE::BYTE, 0, std::ios::beg);
150 
151  // For some reason, some Vicon's file has lot of "0" at the beginning of the file
152  // This part loop up to the point no 0 is found
153  while (!_parametersAddress){
154  _parametersAddress = c3d.readUint(file, 1*ezc3d::DATA_TYPE::BYTE);
155  if (file.eof())
156  throw std::ios_base::failure("File is empty");
157  ++_nbOfZerosBeforeHeader;
158  }
159 
160  _checksum = c3d.readUint(file, 1*ezc3d::DATA_TYPE::BYTE);
161  if (_checksum != 0x50) // If checkbyte is wrong
162  throw std::ios_base::failure("File must be a valid c3d file");
163 
164  // Number of data
165  _nb3dPoints = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
166  _nbAnalogsMeasurement = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
167 
168  // Idx of first and last frame
169  _firstFrame = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD) - 1; // 1-based!
170  _lastFrame = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD) - 1;
171 
172  // Some info
173  _nbMaxInterpGap = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
174  _scaleFactor = c3d.readInt(file, 2*ezc3d::DATA_TYPE::WORD);
175 
176  // Parameters of analog data
177  _dataStart = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
178  _nbAnalogByFrame = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
179  _frameRate = c3d.readFloat(file);
180  _emptyBlock1 = c3d.readInt(file, 135*ezc3d::DATA_TYPE::WORD);
181 
182  // Parameters of keys
183  _keyLabelPresent = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
184  _firstBlockKeyLabel = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
185  _fourCharPresent = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
186 
187  // Parameters of events
188  _nbEvents = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
189  _emptyBlock2 = c3d.readInt(file, 1*ezc3d::DATA_TYPE::WORD);
190  for (unsigned int i = 0; i < _eventsTime.size(); ++i)
191  _eventsTime[i] = c3d.readFloat(file);
192  for (unsigned int i = 0; i < _eventsDisplay.size(); ++i)
193  _eventsDisplay[i] = c3d.readUint(file, 1*ezc3d::DATA_TYPE::WORD);
194  _emptyBlock3 = c3d.readInt(file, 1*ezc3d::DATA_TYPE::WORD);
195  for (unsigned int i = 0; i<_eventsLabel.size(); ++i)
196  _eventsLabel[i] = c3d.readString(file, 2*ezc3d::DATA_TYPE::WORD);
197  _emptyBlock4 = c3d.readInt(file, 22*ezc3d::DATA_TYPE::WORD);
198 }
