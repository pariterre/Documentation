131 int ezc3d::ParametersNS::GroupNS::Parameter::read(ezc3d::c3d &c3d, std::fstream &file, int nbCharInName)
132 {
133  if (nbCharInName < 0)
134  _isLocked = true;
135  else
136  _isLocked = false;
137 
138  // Read name of the group
139  _name = c3d.readString(file, static_cast<unsigned int>(abs(nbCharInName) * ezc3d::DATA_TYPE::BYTE));
140 
141  // number of byte to the next group from here
142  int offsetNext(static_cast<int>(c3d.readUint(file, 2*ezc3d::DATA_TYPE::BYTE)));
143  // Compute the position of the element in the file
144  int nextParamByteInFile;
145  if (offsetNext == 0)
146  nextParamByteInFile = 0;
147  else
148  nextParamByteInFile = static_cast<int>(file.tellg()) + offsetNext - ezc3d::DATA_TYPE::WORD;
149 
150  // -1 sizeof(char), 1 byte, 2 int, 4 float
151  int lengthInByte(c3d.readInt(file, 1*ezc3d::DATA_TYPE::BYTE));
152  if (lengthInByte == -1)
153  _data_type = DATA_TYPE::CHAR;
154  else if (lengthInByte == 1)
155  _data_type = DATA_TYPE::BYTE;
156  else if (lengthInByte == 2)
157  _data_type = DATA_TYPE::INT;
158  else if (lengthInByte == 4)
159  _data_type = DATA_TYPE::FLOAT;
160  else
161  throw std::ios_base::failure ("Parameter type unrecognized");
162 
163  // number of dimension of parameter (0 for scalar)
164  int nDimensions(c3d.readInt(file, 1*ezc3d::DATA_TYPE::BYTE));
165  if (nDimensions == 0 && _data_type != DATA_TYPE::CHAR) // In the special case of a scalar
166  _dimension.push_back(1);
167  else // otherwise it's a matrix
168  for (int i=0; i<nDimensions; ++i)
169  _dimension.push_back (c3d.readUint(file, 1*ezc3d::DATA_TYPE::BYTE)); // Read the dimension size of the matrix
170 
171  // Read the data for the parameters
172  if (_data_type == DATA_TYPE::CHAR)
173  c3d.readParam(file, _dimension, _param_data_string);
174  else if (_data_type == DATA_TYPE::BYTE)
175  c3d.readParam(file, static_cast<unsigned int>(_data_type), _dimension, _param_data_int);
176  else if (_data_type == DATA_TYPE::INT)
177  c3d.readParam(file, static_cast<unsigned int>(_data_type), _dimension, _param_data_int);
178  else if (_data_type == DATA_TYPE::FLOAT)
179  c3d.readParam(file, _dimension, _param_data_float);
180 
181 
182  // Byte 5+nbCharInName ==> Number of characters in group description
183  int nbCharInDesc(c3d.readInt(file, 1*ezc3d::DATA_TYPE::BYTE));
184  // Byte 6+nbCharInName ==> Group description
185  if (nbCharInDesc)
186  _description = c3d.readString(file, static_cast<unsigned int>(nbCharInDesc));
187 
188  // Return how many bytes
189  return nextParamByteInFile;
190 }
