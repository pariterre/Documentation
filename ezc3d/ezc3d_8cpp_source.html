462  }

466  _header->frameRate(pointRate);
467  }
468  if (static_cast<size_t>(parameters().group("POINT").parameter("USED").valuesAsInt()[0]) != header().nb3dPoints()){
469  _header->nb3dPoints(static_cast<size_t>(parameters().group("POINT").parameter("USED").valuesAsInt()[0]));
470  }
471 
472  // Compare the subframe with data when possible, otherwise go with the parameters
473  if (_data != nullptr && data().nbFrames() > 0 && data().frame(0).analogs().nbSubframes() != 0) {
474  if (data().frame(0).analogs().nbSubframes() != static_cast<size_t>(header().nbAnalogByFrame()))
475  _header->nbAnalogByFrame(data().frame(0).analogs().nbSubframes());
476  } else {
477  if (static_cast<size_t>(pointRate) == 0){
478  if (static_cast<size_t>(header().nbAnalogByFrame()) != 1)
479  _header->nbAnalogByFrame(1);

482  _header->nbAnalogByFrame(static_cast<size_t>(parameters().group("ANALOG").parameter("RATE").valuesAsFloat()[0] / pointRate));
483  }
484  }
