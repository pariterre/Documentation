165 ezc3d::ParametersNS::Parameters::Parameters(ezc3d::c3d &file) :
166  _parametersStart(0),
167  _checksum(0),
168  _nbParamBlock(0),
169  _processorType(0)
170 {
171  // Read the Parameters Header
172  _parametersStart = file.readUint(1*ezc3d::DATA_TYPE::BYTE, static_cast<int>(256*ezc3d::DATA_TYPE::WORD*(file.header().parametersAddress()-1) + file.header().nbOfZerosBeforeHeader()), std::ios::beg);
173  _checksum = file.readUint(1*ezc3d::DATA_TYPE::BYTE);
174  _nbParamBlock = file.readUint(1*ezc3d::DATA_TYPE::BYTE);
175  _processorType = file.readUint(1*ezc3d::DATA_TYPE::BYTE);
176  if (_checksum == 0 && _parametersStart == 0){
177  // Theoritically, if this happens, this is a bad c3d formatting and should return an error, but for some reason
178  // Qualisys decided that they would not comply to the standard. Therefore they put "_parameterStart" and "_checksum" to 0
179  // This is a patch for Qualisys bad formatting c3d
180  _parametersStart = 1;
181  _checksum = 0x50;
182  }
183  if (_checksum != 0x50) // If checkbyte is wrong
184  throw std::ios_base::failure("File must be a valid c3d file");
185 
186  // Read parameter or group
187  std::streampos nextParamByteInFile(static_cast<int>(file.tellg()) + static_cast<int>(_parametersStart) - ezc3d::DATA_TYPE::BYTE);
188  while (nextParamByteInFile)
189  {
190  // Check if we spontaneously got to the next parameter. Otherwise c3d is messed up
191  if (file.tellg() != nextParamByteInFile)
192  throw std::ios_base::failure("Bad c3d formatting");
193 
194  // Nb of char in the group name, locked if negative, 0 if we finished the section
195  int nbCharInName(file.readInt(1*ezc3d::DATA_TYPE::BYTE));
196  if (nbCharInName == 0)
197  break;
198  int id(file.readInt(1*ezc3d::DATA_TYPE::BYTE));
199 
200  // Make sure there at least enough group
201  for (size_t i = _groups.size(); i < static_cast<size_t>(abs(id)); ++i)
202  _groups.push_back(ezc3d::ParametersNS::GroupNS::Group());
203 
204  // Group ID always negative for groups and positive parameter of group ID
205  if (id < 0)
206  nextParamByteInFile = group_nonConst(static_cast<size_t>(abs(id)-1)).read(file, nbCharInName);
207  else
208  nextParamByteInFile = group_nonConst(static_cast<size_t>(id-1)).parameter(file, nbCharInName);
209  }
210 }

227 void ezc3d::ParametersNS::Parameters::write(std::fstream &f) const
228 {
229  // Write the header of parameters
230  f.write(reinterpret_cast<const char*>(&_parametersStart), ezc3d::BYTE);
231  int checksum(0x50);
232  f.write(reinterpret_cast<const char*>(&checksum), ezc3d::BYTE);
233  // Leave a blank space which will be later fill
234  // (number of block can't be known before writing them)
235  std::streampos pos(f.tellg()); // remember where to input this value later
236  int blankValue(0);
237  f.write(reinterpret_cast<const char*>(&blankValue), ezc3d::BYTE);
238  int processorType = 84;
239  f.write(reinterpret_cast<const char*>(&processorType), ezc3d::BYTE);
240 
241  // Write each groups
242  std::streampos dataStartPosition; // Special parameter in POINT group
243  for (size_t i=0; i < nbGroups(); ++i)
244  group(i).write(f, -static_cast<int>(i+1), dataStartPosition);
245 
246  // Move the cursor to a beginning of a block
247  std::streampos actualPos(f.tellg());
248  for (int i=0; i<512 - static_cast<int>(actualPos) % 512; ++i){
249  f.write(reinterpret_cast<const char*>(&blankValue), ezc3d::BYTE);
250  }
251  // Go back at the left blank space and write the actual position
252  actualPos = f.tellg();
253  f.seekg(pos);
254  int nBlocksToNext = int(actualPos - pos-2)/512;
255  if (int(actualPos - pos-2) % 512 > 0)
256  ++nBlocksToNext;
257  f.write(reinterpret_cast<const char*>(&nBlocksToNext), ezc3d::BYTE);
258  f.seekg(actualPos);
259 
260  // Go back to data start blank space and write the actual position
261  actualPos = f.tellg();
262  f.seekg(dataStartPosition);
263  nBlocksToNext = int(actualPos)/512;
264  if (int(actualPos) % 512 > 0)
265  ++nBlocksToNext;
266  f.write(reinterpret_cast<const char*>(&nBlocksToNext), ezc3d::BYTE);
267  f.seekg(actualPos);
268 }
