129 double biorbd::actuator::ActuatorGauss3p::torqueMax(
130  const biorbd::rigidbody::GeneralizedCoordinates &Q,
131  const biorbd::rigidbody::GeneralizedVelocity &Qdot){
132  double pos(Q[*m_dofIdx] * 180/M_PI);
133  double speed(Qdot[*m_dofIdx] * 180/M_PI);
134 
135  // Tetanic torque max
136  double Tc = *m_T0 * *m_wc / *m_wmax;
137  double C = Tc * (*m_wmax + *m_wc); // concentric
138  double we = ( (*m_Tmax - *m_T0) * *m_wmax * *m_wc ) / ( *m_k * *m_T0 * (*m_wmax + *m_wc) );
139  double E = -( *m_Tmax - *m_T0 ) * we; // excentric
140 
141  double Tw;
142  if (speed >= 0)
143  Tw = C / ( *m_wc + speed ) - Tc; // For the concentric
144  else
145  Tw = E / ( we - speed ) + *m_Tmax; // For the excentric
146 
147 
148  // Differential activation
149  double A = *m_amin + ( *m_amax - *m_amin ) / ( 1 + exp( -(speed - *m_w1) / *m_wr ) );
150 
151  // Torque angle
152  double Ta = exp( -(*m_qopt - pos) * (*m_qopt - pos) / (2 * *m_r * *m_r) );
153 
154  // Calculation of the max torque
155  return Tw * A * Ta;
156 
157 }
