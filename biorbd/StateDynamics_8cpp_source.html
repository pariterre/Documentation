70  const biorbd::muscles::Characteristics &characteristics,
71  bool alreadyNormalized){
72  // Implémentation de la fonction da/dt = (u-a)/GeneralizedTorque(u,a)
73  // ou GeneralizedTorque(u,a) = t_act(0.5+1.5*a) is u>a et GeneralizedTorque(u,a)=t_deact(0.5+1.5*a) sinon
74  if (*m_activation<characteristics.minActivation())
75  *m_activation = characteristics.minActivation();
76 
77  if (*m_excitation<characteristics.minActivation())
78  *m_excitation = characteristics.minActivation();
79 
80 
81 
82 // see doi:10.1016/j.humov.2011.08.006
83 // see doi:10.1016/S0021-9290(03)00010-1
84 
85 
86 // http://simtk-confluence.stanford.edu:8080/display/OpenSim/First-Order+Activation+Dynamics
87 
88  double num;
89  if (alreadyNormalized)
90  num = *m_excitation- *m_activation; // numérateur
91  else
92  num = normalizeExcitation(characteristics.stateMax())- *m_activation; // numérateur
93 
94  double denom; // dénominateur
95  if (num>0)
96  denom = characteristics.torqueActivation() * (0.5+1.5* *m_activation);
97  else
98  denom = characteristics.torqueDeactivation() / (0.5+1.5* *m_activation);
99 
100  *m_activationDot = num/denom;
101 
102  return *m_activationDot;
103 }
