47 biorbd::utils::Rotation& biorbd::utils::Rotation::fromEulerAngles(
48  const biorbd::utils::Vector &rot,
49  const biorbd::utils::String& seq)
50 {
51  // Check for size consistency
52  biorbd::utils::Error::check(
53  seq.length() == static_cast<unsigned int>(rot.rows()),
54  "Rotation and sequence of rotation must be the same length");
55 
56  setIdentity();
57  // Set the actual rotation matrix to this
58  Eigen::Matrix3d tp;
59  for (unsigned int i=0; i<seq.length(); ++i){
60  double cosVi(std::cos(rot[i]));
61  double sinVi(std::sin(rot[i]));
62  if (seq.tolower()[i] == 'x')
63  tp << 1, 0, 0,
64  0, cosVi, -sinVi,
65  0, sinVi, cosVi;
66 
67  else if (seq.tolower()[i] == 'y')
68  tp << cosVi, 0, sinVi,
69  0, 1, 0,
70  -sinVi, 0, cosVi;
71 
72  else if (seq.tolower()[i] == 'z')
73  tp << cosVi, -sinVi, 0,
74  sinVi, cosVi, 0,
75  0, 0, 1;
76  else
77  biorbd::utils::Error::raise("Rotation sequence not recognized");
78 
79  block(0,0,3,3) *= tp;
80  }
81  return *this;
82 }

84 biorbd::utils::Rotation biorbd::utils::Rotation::fromMarkers(
85  const std::pair<biorbd::rigidbody::NodeSegment, biorbd::rigidbody::NodeSegment> &axis1markers,
86  const std::pair<biorbd::rigidbody::NodeSegment, biorbd::rigidbody::NodeSegment> &axis2markers,
87  const std::pair<biorbd::utils::String, biorbd::utils::String>& axesNames,
88  const biorbd::utils::String &axisToRecalculate)
89 {
90  // Figure out where to put the axes
91  std::vector<unsigned int> map(3);
92  std::vector<unsigned int> toMultiply(2);
93  if (!axesNames.first.tolower().compare("x")){
94  map[0] = 0;
95  if (!axesNames.second.tolower().compare("y")){
96  map[1] = 1;
97  map[2] = 2;
98  toMultiply[0] = 0;
99  toMultiply[1] = 1;
100  } else {
101  map[1] = 2;
102  map[2] = 1;
103  toMultiply[0] = 2;
104  toMultiply[1] = 0;
105  }
106  }
107  else if (!axesNames.first.tolower().compare("y")){
108  map[0] = 1;
109  if (!axesNames.second.tolower().compare("x")){
110  map[1] = 0;
111  map[2] = 2;
112  toMultiply[0] = 0;
113  toMultiply[1] = 1;
114  } else {
115  map[1] = 2;
116  map[2] = 0;
117  toMultiply[0] = 1;
118  toMultiply[1] = 2;
119  }
120  }
121  else if (!axesNames.first.tolower().compare("z")){
122  map[0] = 2;
123  if (!axesNames.second.tolower().compare("x")){
124  map[1] = 0;
125  map[2] = 1;
126  toMultiply[0] = 2;
127  toMultiply[1] = 0;
128  } else {
129  map[1] = 1;
130  map[2] = 0;
131  toMultiply[0] = 1;
132  toMultiply[1] = 2;
133  }
134  }
135 
136  // Get the system of axis XYZ
137  std::vector<biorbd::utils::Vector3d> axes(3);
138  axes[map[0]] = axis1markers.second - axis1markers.first;
139  axes[map[1]] = axis2markers.second - axis2markers.first;
140  axes[map[2]] = axes[toMultiply[0]].cross(axes[toMultiply[1]]);
141 
142  // Recalculate one axis
143  if (!axisToRecalculate.tolower().compare("x")){
144  axes[0] = axes[1].cross(axes[2]);
145  } else if (!axisToRecalculate.tolower().compare("y")){
146  axes[1] = axes[2].cross(axes[0]);
147  } else if (!axisToRecalculate.tolower().compare("z")){
148  axes[2] = axes[0].cross(axes[1]);
149  }
150 
151  // Organize them in a normalized matrix
152  biorbd::utils::Rotation r_out;
153  for (unsigned int i=0; i<3; ++i){
154  // Normalize axes
155  axes[i].normalize();
156  r_out.block(0, i, 3, 1) = axes[i];
157  }
158 
159  return r_out;
160 }

162 biorbd::utils::Vector biorbd::utils::Rotation::toEulerAngles(
163  const biorbd::utils::Rotation &r,
164  const biorbd::utils::String &seq)
165 {
166  biorbd::utils::Vector v;
167  if (!seq.compare("zyzz"))
168  v = biorbd::utils::Vector(3);
169  else
170  v = biorbd::utils::Vector(static_cast<unsigned int>(seq.length()));
171 
172  if (!seq.compare("x")) {
173  v[0] = asin(r(2, 1)); // x
174  }
175  else if (!seq.compare("y")) {
176  v[0] = asin(r(0, 2)); // y
177  }
178  else if (!seq.compare("z")) {
179  v[0] = asin(r(1, 0)); // z
180  }
181  else if (!seq.compare("xy")) {
182  v[0] = asin(r(2,1)); // x
183  v[1] = asin(r(0,2)); // y
184  }
185  else if (!seq.compare("xz")) {
186  v[0] = -asin(r(1,2)); // x
187  v[1] = -asin(r(0,1)); // z
188  }
189  else if (!seq.compare("yx")) {
190  v[0] = -asin(r(2,0)); // y
191  v[1] = -asin(r(1,2)); // x
192  }
193  else if (!seq.compare("yz")) {
194  v[0] = asin(r(0,2)); // y
195  v[1] = asin(r(1,0)); // z
196  }
197  else if (!seq.compare("zx")) {
198  v[0] = asin(r(1,0)); // z
199  v[1] = asin(r(2,1)); // x
200  }
201  else if (!seq.compare("zy")) {
202  v[0] = -asin(r(0,1)); // z
203  v[1] = -asin(r(2,0)); // y
204  }
205  else if (!seq.compare("xyz")) {
206  v[0] = atan2(-r(1,2), r(2,2)); // x
207  v[1] = asin(r(0,2)); // y
208  v[2] = atan2(-r(0,1), r(0,0)); // z
209  }
210  else if (!seq.compare("xzy")) {
211  v[0] = atan2(r(2,1), r(1,1)); // x
212  v[1] = asin(-r(0,1)); // z
213  v[2] = atan2(r(0,2), r(0,0)); // y
214  }
215  else if (!seq.compare("yxz")) {
216  v[0] = atan2(r(0,2), r(2,2)); // y
217  v[1] = asin(-r(1,2)); // x
218  v[2] = atan2(r(1,0), r(1,1)); // z
219  }
220  else if (!seq.compare("yzx")) {
221  v[0] = atan2(-r(2,0), r(0,0)); // y
222  v[1] = asin(r(1,0)); // z
223  v[2] = atan2(-r(1,2), r(1,1)); // x
224  }
225  else if (!seq.compare("zxy")) {
226  v[0] = atan2(-r(0,1), r(1,1)); // z
227  v[1] = asin(r(2,1)); // x
228  v[2] = atan2(-r(2,0), r(2,2)); // y
229  }
230  else if (!seq.compare("zyx")) {
231  v[0] = atan2(r(1,0), r(0,0)); // z
232  v[1] = asin(-r(2,0)); // y
233  v[2] = atan2(r(2,1), r(2,2)); // x
234  }
235  else if (!seq.compare("zxz")) {
236  v[0] = atan2(r(0,2), -r(1,2)); // z
237  v[1] = acos(r(2,2)); // x
238  v[2] = atan2(r(2,0), r(2,1)); // z
239  }
240  else if (!seq.compare("zyz")) {
241  v[0] = atan2(r(1,2), r(0,2)); // z
242  v[1] = acos(r(2,2)); // y
243  v[2] = atan2(r(2,1), -r(2,0)); // z
244  }
245  else if (!seq.compare("zyzz")) {
246  v[0] = atan2(r(1,2), r(0,2)); // z
247  v[1] = acos(r(2,2)); // y
248  v[2] = atan2(r(2,1), -r(2,0)) + v[0]; // z+z
249  }
250  else {
251  biorbd::utils::Error::raise("Angle sequence is not recognized");
252  }
253 
254  return v;
255 }

258  const std::vector<biorbd::utils::Rotation> & mToMean)
259 {

262 
263  // Initial guess being the actual arithmetic mean
264  for (unsigned int i = 0; i<mToMean.size(); ++i){
265  m_tp += mToMean[i];
266  }
267  m_tp = m_tp/mToMean.size();
268 
269  // SVD decomposition
270  Eigen::JacobiSVD<Eigen::Matrix3d> svd(
271  m_tp, Eigen::ComputeFullU | Eigen::ComputeFullV);
272 
273  // Normalize the matrix
274  biorbd::utils::Rotation m_out(svd.matrixU() * svd.matrixV().transpose());
275
