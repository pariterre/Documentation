266 void biorbd::muscles::HillType::computeFlCE(const biorbd::muscles::StateDynamics &emg){
267  *m_FlCE = exp( -pow(( position().length() / m_characteristics->optimalLength() / (*m_cste_FlCE_1*(1-emg.activation())+1) -1 ), 2)
268  /
269  *m_cste_FlCE_2 );

272 void biorbd::muscles::HillType::computeFvCE(){
273  // The relation is different if velocity< 0 or > 0
274  double v = m_position->velocity();
275  if (v<=0)
276  *m_FvCE = (1-abs(v) / *m_cste_maxShorteningSpeed) /
277  (1+abs(v) / *m_cste_maxShorteningSpeed / *m_cste_FvCE_1);
278  else
279  *m_FvCE = (1-1.33*v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2) /
280  (1-v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2);
281 }

283 void biorbd::muscles::HillType::computeFlPE(){
284  if (position().length() > characteristics().tendonSlackLength())
285  *m_FlPE = exp(*m_cste_FlPE_1*(position().length()/characteristics().optimalLength()-1) - *m_cste_FlPE_2);
286  else
287  *m_FlPE = 0;
288 }
