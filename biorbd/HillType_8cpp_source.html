267  *m_FlCE = exp( -pow(( position().length() / m_characteristics->optimalLength() / (*m_cste_FlCE_1*(1-emg.activation())+1) -1 ), 2)
268  /
269  *m_cste_FlCE_2 );

272 void biorbd::muscles::HillType::computeFvCE(){
273  // The relation is different if velocity< 0 or > 0
274  biorbd::utils::Scalar v = m_position->velocity();
275 #ifdef BIORBD_USE_CASADI_MATH
276  *m_FvCE = casadi::MX::if_else(
277  casadi::MX::le(v, 0),
278  (1.0-std::fabs(v) / *m_cste_maxShorteningSpeed) /
279  (1.0+std::fabs(v) / *m_cste_maxShorteningSpeed / *m_cste_FvCE_1),
280  (1.0-1.33*v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2) /
281  (1-v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2)
282  );
283 #else
284  if (v<=0)
285  *m_FvCE = (1-fabs(v) / *m_cste_maxShorteningSpeed) /
286  (1+fabs(v) / *m_cste_maxShorteningSpeed / *m_cste_FvCE_1);
287  else
288  *m_FvCE = (1-1.33*v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2) /
289  (1-v / *m_cste_maxShorteningSpeed / *m_cste_FvCE_2);
290 #endif
291 }

293 void biorbd::muscles::HillType::computeFlPE(){
294 
295 #ifdef BIORBD_USE_CASADI_MATH
296  *m_FlPE = casadi::MX::if_else_zero(
297  casadi::MX::gt(position().length(), characteristics().tendonSlackLength()),
298  exp(*m_cste_FlPE_1*(position().length()/characteristics().optimalLength()-1) - *m_cste_FlPE_2));
299 #else
300  if (position().length() > characteristics().tendonSlackLength())
301  *m_FlPE = exp(*m_cste_FlPE_1*(position().length()/characteristics().optimalLength()-1) - *m_cste_FlPE_2);
302  else
303  *m_FlPE = 0;
304 #endif
305 }
