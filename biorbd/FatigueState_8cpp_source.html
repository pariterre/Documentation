70 void biorbd::muscles::FatigueState::setState(
71  biorbd::utils::Scalar active,
72  biorbd::utils::Scalar fatigued,
73  biorbd::utils::Scalar resting,
74  bool turnOffWarnings)
75 {
76  // Sanity check for active fibers
77  //
78  // In order to get the quantity of active fibers to 0 or 1, it has to come from the input command.
79  // The input command manage fiber recruitment from resting to active.
80  // Hence any exceeding because of integration can be corrected by putting back exceeding active fibers quantity
81  // into resting fibers quantity.
82  //
83  if (active < 0){
84  if (!turnOffWarnings) {
85  biorbd::utils::Error::warning(0,
86  "Active Fibers Quantity can't be lower than 0, 0 is used then\n"
87  "Previous Active Fibers Quantity before set to 0:"
88  + std::to_string(active));
89  }
90  resting += active;
91  active = 0;
92  }
93  else if (active > 1){
94  if (!turnOffWarnings) {
95  biorbd::utils::Error::warning(0,
96  "Active Fibers Quantity can't be higher than 1, 1 is used then\n"
97  "Previous Active Fibers Quantity before set to 1: "
98  + std::to_string(active));
99  }
100  resting += active - 1;
101 
102  active = 1;
103  }
104 
105  // Sanity check for fatigued fibers
106  if (fatigued < 0){
107  biorbd::utils::Error::raise("Fatigued Fibers Quantity can't be lower than 0");
108  }
109  else if (fatigued > 1){
110  biorbd::utils::Error::raise("Fatigued Fibers Quantity can't be higher than 1");
111  }
112 
113  // Sanity check for resting fibers
114  //
115  // In order to get the quantity of resting fibers to 0, it has to come from the input command.
116  // The input command manage fiber recruitment from resting to active.
117  // Hence any exceeding because of integration can be corrected by putting back exceeding resting fibers quantity
118  // into active fibers quantity.
119  //
120  if (resting < 0){
121  if (!turnOffWarnings) {
122  biorbd::utils::Error::warning(0,
123  "Resting Fibers Quantity can't be lower than 0, 0 is used then\n"
124  "Previous Resting Fibers Quantity before set to 0: "
125  + std::to_string(resting));
126  }
127  active += resting;
128  resting = 0;
129  }
130  else if (resting > 1){
131  biorbd::utils::Error::raise(
132  "Resting Fibers Quantity can't be higher than 1");
133 
134  }
135 
136  if (fabs(active + fatigued + resting - 1.0) > 0.1){
137  biorbd::utils::Error::raise("Sum of the fatigued states must be equal to 1");
138  }
139 
140  *m_activeFibers = active;
141  *m_fatiguedFibers = fatigued;
142  *m_restingFibers = resting;
143 }
