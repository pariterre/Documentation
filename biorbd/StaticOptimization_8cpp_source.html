179 void biorbd::muscles::StaticOptimization::run(
180  bool useLinearizedState)
181 {
182  // Setup the Ipopt problem
183  Ipopt::SmartPtr<Ipopt::IpoptApplication> app = IpoptApplicationFactory();
184  app->Options()->SetNumericValue("tol", 1e-7);
185  app->Options()->SetStringValue("mu_strategy", "adaptive");
186  //app->Options()->SetStringValue("output_file", "ipopt.out");
187  app->Options()->SetStringValue("hessian_approximation", "limited-memory");
188  app->Options()->SetStringValue("derivative_test", "first-order");
189  app->Options()->SetIntegerValue("max_iter", 10000);
190 
191  Ipopt::ApplicationReturnStatus status;
192  status = app->Initialize();
193  biorbd::utils::Error::check(status == Ipopt::Solve_Succeeded, "Ipopt initialization failed");
194 
195  for (unsigned int i=0; i<m_allQ->size(); ++i){
196  if (useLinearizedState)
197  m_staticOptimProblem->push_back(
198  new biorbd::muscles::StaticOptimizationIpoptLinearized(
199  m_model, (*m_allQ)[i], (*m_allQdot)[i], (*m_allTorqueTarget)[i], *m_initialActivationGuess,
200  *m_useResidualTorque, *m_pNormFactor, *m_verbose
201  )
202  );
203  else
204  m_staticOptimProblem->push_back(
205  new biorbd::muscles::StaticOptimizationIpopt(
206  m_model, (*m_allQ)[i], (*m_allQdot)[i], (*m_allTorqueTarget)[i], *m_initialActivationGuess,
207  *m_useResidualTorque, *m_pNormFactor, *m_verbose
208  )
209  );
210  // Optimize!
211  status = app->OptimizeTNLP((*m_staticOptimProblem)[i]);
212 
213  // Take the solution of the previous optimization as the solution for the next optimization
214  *m_initialActivationGuess = static_cast<biorbd::muscles::StaticOptimizationIpopt*>(Ipopt::GetRawPtr((*m_staticOptimProblem)[i]))->finalSolution();
215  }
216  *m_alreadyRun = true;
217 }
