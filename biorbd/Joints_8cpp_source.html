912 
913  // Compute H
914  CS.H = RigidBodyDynamics::Math::MatrixNd::Zero(this->dof_count, this->dof_count);
915  RigidBodyDynamics::CompositeRigidBodyAlgorithm (*this, Q, CS.H, false);

928 
929  // only compute the matrix Gi if actually needed
930  if (prev_body_id != CS.body[i] || prev_body_point != CS.point[i]) {

934  }

946  RigidBodyDynamics::Math::Vector3d gamma_i = RigidBodyDynamics::Math::Vector3d::Zero();
947 
948  // update Kinematics just once
949  UpdateKinematics (*this, Q, QDot, CS.QDDot_0);
950

1023 #endif

1027

1030  QDDot[i] = CS.x[i];
