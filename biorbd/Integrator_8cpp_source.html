50  const state_type &x ,
51  state_type &dxdt ,
52  double ){
53 
54  // Équation différentielle : x/xdot => xdot/xddot
55  biorbd::rigidbody::GeneralizedCoordinates Q(*m_model);
56  biorbd::rigidbody::GeneralizedVelocity QDot(*m_model);
57  biorbd::rigidbody::GeneralizedAcceleration QDDot(*m_model);
58  QDDot.setZero();
59  for (unsigned int i=0; i<*m_nQ; i++){
60  Q(i) = x[i];
61  }
62  for (unsigned int i=0; i<*m_nQdot; i++){
63  QDot(i) = x[i+*m_nQ];
64  }
65 
66  RigidBodyDynamics::ForwardDynamics (*m_model, Q, QDot, *m_u, QDDot);
67 
68  // Faire sortir xdot/xddot
69  for (unsigned int i=0; i<*m_nQ; i++){
70  dxdt[i] = QDot[i];
71  }
72  for (unsigned int i=0; i<*m_nQdot; i++){
73  dxdt[i + *m_nQ] = QDDot[i];
74  }
