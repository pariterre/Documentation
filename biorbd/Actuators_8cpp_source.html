82 void biorbd::actuator::Actuators::addActuator(const biorbd::actuator::Actuator &act)
83 {
84  biorbd::utils::Error::check(
85  !*m_isClose, "You can't add actuator after closing the model");
86 
87  // Assuming that this is also a Joints type (via BiorbdModel)
88  const biorbd::rigidbody::Joints &model = dynamic_cast<biorbd::rigidbody::Joints &>(*this);
89 
90  // Verify that the target dof is associated to a dof that
91  // already exists in the model
92  biorbd::utils::Error::check(
93  act.index()<model.nbDof(), "Sent index is out of dof range");
94 
95  // For speed purposes and coherence with the Q,
96  // set the actuator to the same index as its associated dof
97  unsigned int idx(act.index());
98 
99  // If there are less actuators declared than dof,
100  // the vector must be enlarged
101  if (idx >= m_all->size()){
102  m_all->resize(idx+1);
103  m_isDofSet->resize((idx+1)*2, false);
104  }
105 
106  // Add an actuator to the pool of actuators according to its type
107  if (act.type() == biorbd::actuator::TYPE::CONSTANT){
108  if (act.direction() == 1){
109  (*m_all)[idx].first = std::make_shared<biorbd::actuator::ActuatorConstant>(static_cast<const biorbd::actuator::ActuatorConstant&>(act));
110  (*m_isDofSet)[idx*2] = true;
111  }
112  else{
113  (*m_all)[idx].second = std::make_shared<biorbd::actuator::ActuatorConstant>(static_cast<const biorbd::actuator::ActuatorConstant&>(act));
114  (*m_isDofSet)[idx*2+1] = true;
115  }
116  return;
117  }
118  else if (act.type() == biorbd::actuator::TYPE::LINEAR){
119  if (act.direction() == 1){
120  (*m_all)[idx].first = std::make_shared<biorbd::actuator::ActuatorLinear>(static_cast<const biorbd::actuator::ActuatorLinear&>(act));
121  (*m_isDofSet)[idx*2] = true;
122  }
123  else{
124  (*m_all)[idx].second = std::make_shared<biorbd::actuator::ActuatorLinear>(static_cast<const biorbd::actuator::ActuatorLinear&>(act));
125  (*m_isDofSet)[idx*2+1] = true;
126  }
127  return;
128  }
129  else if (act.type() == biorbd::actuator::TYPE::GAUSS3P){
130  if (act.direction() == 1){
131  (*m_all)[idx].first = std::make_shared<biorbd::actuator::ActuatorGauss3p>(static_cast<const biorbd::actuator::ActuatorGauss3p&>(act));
132  (*m_isDofSet)[idx*2] = true;
133  }
134  else {
135  (*m_all)[idx].second = std::make_shared<biorbd::actuator::ActuatorGauss3p>(static_cast<const biorbd::actuator::ActuatorGauss3p&>(act));
136  (*m_isDofSet)[idx*2+1] = true;
137  }
138  return;
139  }
140  else if (act.type() == biorbd::actuator::TYPE::GAUSS6P){
141  if (act.direction() == 1){
142  (*m_all)[idx].first = std::make_shared<biorbd::actuator::ActuatorGauss6p>(static_cast<const biorbd::actuator::ActuatorGauss6p&>(act));
143  (*m_isDofSet)[idx*2] = true;

198 biorbd::rigidbody::GeneralizedTorque biorbd::actuator::Actuators::torque(
199  const biorbd::utils::Vector& activation,
200  const biorbd::rigidbody::GeneralizedCoordinates& Q,
201  const biorbd::rigidbody::GeneralizedVelocity &Qdot)
202 {
203  // Set qdot to be positive if concentric and negative is excentric
204  biorbd::rigidbody::GeneralizedVelocity QdotResigned(Qdot);
205  for (unsigned int i=0; i<Qdot.size(); ++i)
206  if (activation(i)<0)
207  QdotResigned(i) = -Qdot(i);
208 
209  // Calculate the maximal torques
210  biorbd::rigidbody::GeneralizedTorque GeneralizedTorque(
211  torqueMax(activation,Q,QdotResigned));
212 
213  // Put the signs
214  for (unsigned int i=0; i<GeneralizedTorque.size(); ++i)
215  GeneralizedTorque(i) *= activation(i);
216 
217  return GeneralizedTorque;
218 }

226  biorbd::utils::Error::check(*m_isClose, "Close the actuator model before calling torqueMax");
227 
228  // Assuming that this is also a Joints type (via BiorbdModel)
229  const biorbd::rigidbody::Joints &model = dynamic_cast<biorbd::rigidbody::Joints &>(*this);
230 
231  std::pair<biorbd::rigidbody::GeneralizedTorque, biorbd::rigidbody::GeneralizedTorque> maxGeneralizedTorque_all =
232  std::make_pair(biorbd::rigidbody::GeneralizedTorque(model), biorbd::rigidbody::GeneralizedTorque(model));
233 
234  for (unsigned int i=0; i<model.nbDof(); ++i){
235  std::pair<std::shared_ptr<Actuator>, std::shared_ptr<Actuator>> GeneralizedTorque_tp(actuator(i));
236  for (unsigned p=0; p<2; ++p){
237  if (p==0) // First
238  if (std::dynamic_pointer_cast<ActuatorGauss3p> (GeneralizedTorque_tp.first))
239  maxGeneralizedTorque_all.first[i] = std::static_pointer_cast<ActuatorGauss3p> (GeneralizedTorque_tp.first)->torqueMax(Q, Qdot);
240  else if (std::dynamic_pointer_cast<ActuatorConstant> (GeneralizedTorque_tp.first))
241  maxGeneralizedTorque_all.first[i] = std::static_pointer_cast<ActuatorConstant> (GeneralizedTorque_tp.first)->torqueMax();
242  else if (std::dynamic_pointer_cast<ActuatorLinear> (GeneralizedTorque_tp.first))
243  maxGeneralizedTorque_all.first[i] = std::static_pointer_cast<ActuatorLinear> (GeneralizedTorque_tp.first)->torqueMax(Q);
244  else if (std::dynamic_pointer_cast<ActuatorGauss6p> (GeneralizedTorque_tp.first))
245  maxGeneralizedTorque_all.first[i] = std::static_pointer_cast<ActuatorGauss6p> (GeneralizedTorque_tp.first)->torqueMax(Q, Qdot);
246  else
247  biorbd::utils::Error::raise("Wrong type (should never get here because of previous safety)");
248  else // Second
249  if (std::dynamic_pointer_cast<ActuatorGauss3p> (GeneralizedTorque_tp.second))
250  maxGeneralizedTorque_all.second[i] = std::static_pointer_cast<ActuatorGauss3p> (GeneralizedTorque_tp.second)->torqueMax(Q, Qdot);
251  else if (std::dynamic_pointer_cast<ActuatorConstant> (GeneralizedTorque_tp.second))
252  maxGeneralizedTorque_all.second[i] = std::static_pointer_cast<ActuatorConstant> (GeneralizedTorque_tp.second)->torqueMax();
253  else if (std::dynamic_pointer_cast<ActuatorLinear> (GeneralizedTorque_tp.second))
254  maxGeneralizedTorque_all.second[i] = std::static_pointer_cast<ActuatorLinear> (GeneralizedTorque_tp.second)->torqueMax(Q);
255  else if (std::dynamic_pointer_cast<ActuatorGauss6p> (GeneralizedTorque_tp.second))
256  maxGeneralizedTorque_all.second[i] = std::static_pointer_cast<ActuatorGauss6p> (GeneralizedTorque_tp.second)->torqueMax(Q, Qdot);
