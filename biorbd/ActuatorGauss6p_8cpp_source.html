149 double biorbd::actuator::ActuatorGauss6p::torqueMax(
150  const biorbd::rigidbody::GeneralizedCoordinates &Q,
151  const biorbd::rigidbody::GeneralizedVelocity &Qdot){
152  double pos(Q[*m_dofIdx] * 180/M_PI);
153  double speed(Qdot[*m_dofIdx] * 180/M_PI);
154 
155  // Tetanic torque max
156  double Tc = *m_T0 * *m_wc / *m_wmax;
157  double C = Tc*(*m_wmax + *m_wc); // concentric
158  double we = ( (*m_Tmax - *m_T0) * *m_wmax * *m_wc ) / ( *m_k * *m_T0 * (*m_wmax + *m_wc) );
159  double E = -( *m_Tmax - *m_T0 ) * we; // eccentric
160 
161  double Tw;
162  if (speed >= 0)
163  Tw = C / ( *m_wc + speed ) - Tc; // For the concentric
164  else
165  Tw = E / ( we - speed ) + *m_Tmax; // For the eccentric
166 
167  // Differential activation
168  double A = *m_amin + ( *m_amax - *m_amin ) / ( 1 + exp( -(speed - *m_w1) / *m_wr ) );
169 
170  // Torque angle
171  double Ta = exp( -(*m_qopt - pos) * (*m_qopt - pos) / (2* *m_r * *m_r ) )
172  + *m_facteur * exp( -(*m_qopt2 - pos) * (*m_qopt2 - pos) / (2 * *m_r2 * *m_r2) );
173 
174  // Calculation of the max torque
175  return Tw * A * Ta;
176 }
