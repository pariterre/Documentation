373 void biorbd::rigidbody::Segment::setJointAxis(){
374  // Definition of the rotation axis
375  biorbd::utils::Vector3d axis[3];
376  axis[0] = biorbd::utils::Vector3d(1,0,0); // axe x
377  axis[1] = biorbd::utils::Vector3d(0,1,0); // axe y
378  axis[2] = biorbd::utils::Vector3d(0,0,1); // axe z
379 
380  // Declaration of DoFs in translation
381  m_dof->clear();
382  if (*m_nbDof != 0){
383  m_dof->resize(*m_nbDof);
384  for (unsigned int i=0; i<*m_nbDofTrans; i++)
385  (*m_dof)[i] = RigidBodyDynamics::Joint(
386  RigidBodyDynamics::JointTypePrismatic,
387  axis[(*m_dofPosition)[i]]);
388 
389  // Declaration of the DoFs in rotation
390  if (*m_isQuaternion)
391  (*m_dof)[*m_nbDofTrans] = RigidBodyDynamics::Joint(
392  RigidBodyDynamics::JointTypeSpherical);
393  else
394  for (unsigned int i=*m_nbDofTrans; i<*m_nbDofRot+*m_nbDofTrans; i++)
395  (*m_dof)[i] = RigidBodyDynamics::Joint(
396  RigidBodyDynamics::JointTypeRevolute,
397  axis[(*m_dofPosition)[i]]);
398  }
399  else{
400  m_dof->resize(1);
401  (*m_dof)[0] = RigidBodyDynamics::Joint
402  (RigidBodyDynamics::JointTypeFixed);
403  }

406 void biorbd::rigidbody::Segment::setJoints(
407  biorbd::rigidbody::Joints& model) {
408  setDofCharacteristicsOnLastBody(); // Apply the segment caracteristics only to the last segment
409  setJointAxis(); // Choose the axis order in relation to the selected sequence
410 
411 
412  RigidBodyDynamics::Math::SpatialTransform zero (
413  RigidBodyDynamics::Math::Matrix3dIdentity,
414  RigidBodyDynamics::Math::Vector3d(0,0,0));
415  // Create the articulations (intra segment)
416  m_idxDof->clear();
417 
418  if (*m_nbDof==0)
419  m_idxDof->resize(1);
420  else
421  m_idxDof->resize(*m_nbDof);
422 
423  unsigned int parent_id(model.GetBodyId(parent().c_str()));
424  if (parent_id == std::numeric_limits<unsigned int>::max())
425  parent_id = 0;
426  if (*m_nbDof==0)
427  (*m_idxDof)[0] = model.AddBody(
428  parent_id, *m_cor, (*m_dof)[0],
429  (*m_dofCharacteristics)[0], name());
430  else if (*m_nbDof == 1)
431  (*m_idxDof)[0] = model.AddBody(
432  parent_id, *m_cor, (*m_dof)[0],
433  (*m_dofCharacteristics)[0], name());
434  else{
435  (*m_idxDof)[0] = model.AddBody(
436  parent_id, *m_cor, (*m_dof)[0],
437  (*m_dofCharacteristics)[0]);
438  for (unsigned int i=1; i<*m_nbDof; i++)
439  if (i!=*m_nbDof-1)
440  (*m_idxDof)[i] = model.AddBody(
441  (*m_idxDof)[i-1], zero,
442  (*m_dof)[i], (*m_dofCharacteristics)[i]);
443  else
444  (*m_idxDof)[i] = model.AddBody(
445  (*m_idxDof)[i-1], zero, (*m_dof)[i],
446  (*m_dofCharacteristics)[i], name());
