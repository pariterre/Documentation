212 bool biorbd::muscles::StaticOptimizationIpopt::eval_jac_g(
213  Ipopt::Index n,
214  const Ipopt::Number *x,
215  bool new_x,
216  Ipopt::Index m,
217  Ipopt::Index,
218  Ipopt::Index *iRow,
219  Ipopt::Index *jCol,
220  Ipopt::Number *values)
221 {
222  if (values == nullptr) {
223  // Setup non-zeros values
224  Ipopt::Index k(0);
225  for (Ipopt::Index j = 0; static_cast<unsigned int>(j) < *m_nbMus; ++j){
226  for (Ipopt::Index i = 0; i<m; ++i){
227  iRow[k] = i;
228  jCol[k++] = j;
229  }
230  }
231  for (Ipopt::Index j = 0; static_cast<unsigned int>(j) < *m_nbTorqueResidual; ++j ){
232  iRow[k] = j;
233  jCol[k++] = j+ static_cast<Ipopt::Index>(*m_nbMus);
234  }
235  } else {
236  if (new_x)
237  dispatch(x);
238  const biorbd::rigidbody::GeneralizedTorque& GeneralizedTorqueMusc(m_model.muscularJointTorque(*m_states, false, m_Q.get(), m_Qdot.get()));
239  unsigned int k(0);
240  for( unsigned int j = 0; j < *m_nbMus; ++j ){
241  std::vector<std::shared_ptr<biorbd::muscles::StateDynamics>> stateEpsilon;
242  for (unsigned int i = 0; i < *m_nbMus; ++i){
243  unsigned int delta(0);
244  if (i == j)
245  delta = 1;
246  stateEpsilon.push_back(std::make_shared<biorbd::muscles::StateDynamics>(biorbd::muscles::StateDynamics(0, (*m_activations)[i]+delta* *m_eps)));
247  }
248  const biorbd::rigidbody::GeneralizedTorque& GeneralizedTorqueCalculEpsilon(m_model.muscularJointTorque(stateEpsilon, false, m_Q.get(), m_Qdot.get()));
249  for( unsigned int i = 0; i < static_cast<unsigned int>(m); i++ ){
250  values[k++] = (GeneralizedTorqueCalculEpsilon[i]-GeneralizedTorqueMusc[i])/ *m_eps;
251  if (*m_verbose >= 3){
252  std::cout << std::setprecision (20) << std::endl;
253  std::cout << "values[" << k-1 << "]: " << values[k-1] << std::endl;
254  std::cout << "GeneralizedTorqueCalculEpsilon[" << i << "]: " << GeneralizedTorqueCalculEpsilon[i] << std::endl;
255  std::cout << "GeneralizedTorqueMusc[" << i << "]: " << GeneralizedTorqueMusc[i] << std::endl;
256  }
257 
258  }
259  }
260  for( unsigned int j = 0; j < *m_nbTorqueResidual; j++ )
261  values[k++] = 1;
262 
263  if (*m_verbose >= 2){
264  k = 0;
265  biorbd::utils::Matrix jacobian(biorbd::utils::Matrix::Zero(*m_nbTorque, static_cast<unsigned int>(n)));
266  for( unsigned int j = 0; j < *m_nbMus; j++ )
267  for( unsigned int i = 0; i < static_cast<unsigned int>(m); i++ )
268  jacobian(i,j) = values[k++];
269  for( unsigned int j = 0; j < *m_nbTorqueResidual; j++ ){
270  jacobian(j, j+ *m_nbMus) = values[k++];
271  }
272  std::cout << "jacobian = " << std::endl << jacobian << std::endl;
273  }
274 
275  }
276  return true;
