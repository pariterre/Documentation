59 void biorbd::rigidbody::KalmanRecons::iteration(
60  biorbd::utils::Vector measure,
61  const biorbd::utils::Vector &projectedMeasure,
62  const biorbd::utils::Matrix &Hessian,
63  const std::vector<unsigned int> &occlusion){
64  // Prediction
65  const biorbd::utils::Vector& xkm(*m_A * *m_xp);
66  const biorbd::utils::Matrix& Pkm(*m_A * *m_Pp * m_A->transpose() + *m_Q);
67 
68  // Correction
69  biorbd::utils::Matrix InvTp( (Hessian * Pkm * Hessian.transpose() + *m_R).inverse() );
70  manageOcclusionDuringIteration(InvTp, measure, occlusion);
71  const biorbd::utils::Matrix& K(Pkm*Hessian.transpose() * InvTp); // Gain
72 
73  *m_xp = xkm+K*(measure-projectedMeasure); // New estimated state
74  const Eigen::MatrixXd& temp(biorbd::utils::Matrix::Identity(3* *m_nbDof, 3* *m_nbDof) - K*Hessian);
75  *m_Pp = temp * Pkm * temp.transpose() + K* *m_R*K.transpose();
76 
77 }

80  biorbd::utils::Matrix &InvTp,
81  biorbd::utils::Vector &measure,
82  const std::vector<unsigned int> &occlusion){
83  for (unsigned int i = 0; i < occlusion.size(); ++i)
84  for (unsigned int j=occlusion[i]; j< occlusion[i]+1; ++j){
85  InvTp(j,j) = 0; // Artifact due to the fact that m_R has a value at (j:j+2,j:j+2)

105 biorbd::utils::Matrix biorbd::rigidbody::KalmanRecons::evolutionMatrix(
106  const unsigned int nQ,
107  unsigned int n,
108  double Te){
109  // m : number of degrees of freedom
110  // n : order of Taylor development
111  // Te : 1 / (acquisition frequency)
112 
113  n += 1;
114  biorbd::utils::Matrix A(biorbd::utils::Matrix::Identity(nQ*n,nQ*n));
115  double c = 1;
116  for (unsigned int i=2; i<n+1; ++i){
117 
118  unsigned int j=(i-1) * nQ;
119  c = c/(i-1);
120 
121 
122  for (unsigned int cmp=0; cmp<nQ*n-j; ++cmp)
123  A(0+cmp,j+cmp) += c* static_cast<double>(std::pow(Te,(static_cast<double>(i)-1.0)));
124  }
125 
126  return A;
127 }

129 biorbd::utils::Matrix biorbd::rigidbody::KalmanRecons::processNoiseMatrix(
130  const unsigned int nbQ,
131  double Te){
132 
133  // Find the coefficients values
134  double c1 = 1.0/20.0 * pow(Te,5);
135  double c2 = 1.0/8.0 * pow(Te,4);
136  double c3 = 1.0/6.0 * pow(Te,3);
137  double c4 = 1.0/3.0 * pow(Te,3);
138  double c5 = 1.0/2.0 * pow(Te,2);
139  double c6 = Te;
140 
141  // Ouput matrix
142  biorbd::utils::Matrix Q(biorbd::utils::Matrix::Zero(3*nbQ,3*nbQ));
143  for (unsigned int j=0; j<nbQ; ++j){
144  Q( j, j) = c1;
145  Q( j, nbQ+j) = c2;
146  Q( j, 2*nbQ+j) = c3;
147  Q( nbQ+j, j) = c2;
148  Q( nbQ+j, nbQ+j) = c4;
149  Q( nbQ+j, 2*nbQ+j) = c5;
150  Q(2*nbQ+j, j) = c3;
151  Q(2*nbQ+j, nbQ+j) = c5;
152  Q(2*nbQ+j, 2*nbQ+j) = c6;
153  }
154 
155  return Q;
156 }
