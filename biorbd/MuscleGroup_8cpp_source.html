53 {
54  m_mus->resize(other.m_mus->size());

56  if ((*other.m_mus)[i]->type() == biorbd::muscles::MUSCLE_TYPE::IDEALIZED_ACTUATOR)
57  (*m_mus)[i] = std::make_shared<biorbd::muscles::IdealizedActuator>((*other.m_mus)[i]);
58  else if ((*other.m_mus)[i]->type() == biorbd::muscles::MUSCLE_TYPE::HILL)
59  (*m_mus)[i] = std::make_shared<biorbd::muscles::HillType>((*other.m_mus)[i]);
60  else if ((*other.m_mus)[i]->type() == biorbd::muscles::MUSCLE_TYPE::HILL_THELEN)
61  (*m_mus)[i] = std::make_shared<biorbd::muscles::HillThelenType>((*other.m_mus)[i]);
62  else if ((*other.m_mus)[i]->type() == biorbd::muscles::MUSCLE_TYPE::HILL_THELEN_FATIGABLE)
63  (*m_mus)[i] = std::make_shared<biorbd::muscles::HillThelenTypeFatigable>((*other.m_mus)[i]);
64  else
65  biorbd::utils::Error::raise("DeepCopy was not prepared to copy " + biorbd::utils::String(biorbd::muscles::MUSCLE_TYPE_toStr((*other.m_mus)[i]->type())) + " type");
66  }
67  *m_mus = *other.m_mus;
